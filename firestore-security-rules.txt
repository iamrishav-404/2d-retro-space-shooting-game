rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    
    // Players Collection - User-specific data
    match /players/{userId} {
      // Users can only read/write their own data
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Leaderboard Collection - Public read, authenticated write
    match /leaderboard/{scoreId} {
      // Anyone can read the leaderboard
      allow read: if true;
      
      // Only authenticated users can create scores
      allow create: if 
        request.auth != null &&
        request.resource.data.userId == request.auth.uid;
      
      // Users can only update/delete their own scores
      allow update, delete: if 
        request.auth != null && 
        resource.data.userId == request.auth.uid;
    }
    
    // Legacy scores collection (read-only)
    match /scores/{scoreId} {
      allow read: if true;
    }
    
    // Deny all other collections
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
